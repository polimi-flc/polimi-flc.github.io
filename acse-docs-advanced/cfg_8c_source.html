12{

21/* Alloc a new control flow graph register object. If a register object
22 * referencing the same identifier already exists, returns the pre-existing
23 * object. */
24static t_cfgReg *createCFGRegister(t_cfg *graph, t_instrArg *arg)
25{
26 // Test if a register with the same identifier is already present.
27 t_listNode *elementFound =
28 listFindWithCallback(graph->registers, &arg->ID, compareCFGRegAndRegID);
29
30 t_cfgReg *result;
31 if (elementFound) {
32 // It's there: just use it.
33 result = (t_cfgReg *)elementFound->data;
34 } else {
35 // If it's not there it needs to be created.
36 result = malloc(sizeof(t_cfgReg));
37 if (result == NULL)
38 fatalError("out of memory");
39 result->tempRegID = arg->ID;
40 result->mcRegWhitelist = NULL;
41 // Insert it in the list of registers
42 graph->registers = listInsert(graph->registers, result, -1);
43 }
44
45 // Copy the machine register allocation constraint, or compute the
46 // intersection between the register allocation constraint sets.
47 if (arg->mcRegWhitelist) {
48 if (result->mcRegWhitelist == NULL) {
49 result->mcRegWhitelist = listClone(arg->mcRegWhitelist);
50 } else {
51 t_listNode *thisReg = result->mcRegWhitelist;
52 while (thisReg) {
53 t_listNode *nextReg = thisReg->next;
54 if (!listFind(arg->mcRegWhitelist, thisReg->data)) {
55 result->mcRegWhitelist =
56 listRemoveNode(result->mcRegWhitelist, thisReg);
57 }
58 thisReg = nextReg;
59 }
60 if (result->mcRegWhitelist == NULL)
61 fatalError("bug: unsatisfiable register constraints on t%d", arg->ID);
62 }
63 }
64
65 return result;
66}

69static t_bbNode *newBBNode(t_instruction *instr)
70{
71 t_bbNode *result = malloc(sizeof(t_bbNode));
72 if (result == NULL)
73 fatalError("out of memory");
74 for (int i = 0; i < CFG_MAX_DEFS; i++)
75 result->defs[i] = NULL;
76 for (int i = 0; i < CFG_MAX_USES; i++)
77 result->uses[i] = NULL;
78 result->instr = instr;
79 result->in = NULL;
80 result->out = NULL;
81 result->parent = NULL;
82 return result;
83}

96static void bbNodeComputeDefUses(t_bbNode *node)
97{
98 t_cfg *graph = node->parent->parent;
99 t_instruction *instr = node->instr;
100
101 // Create or lookup CFG register objects for all arguments.
102 t_cfgReg *regDest = NULL;
103 t_cfgReg *regSource1 = NULL;
104 t_cfgReg *regSource2 = NULL;
105 if (instr->rDest != NULL)
106 regDest = createCFGRegister(graph, instr->rDest);
107 if (instr->rSrc1 != NULL)
108 regSource1 = createCFGRegister(graph, instr->rSrc1);
109 if (instr->rSrc2 != NULL)
110 regSource2 = createCFGRegister(graph, instr->rSrc2);
111
112 // Fill the def/use sets for this node.
113 int defIdx = 0;
114 if (regDest)
115 node->defs[defIdx++] = regDest;
116 int useIdx = 0;
117 if (regSource1)
118 node->uses[useIdx++] = regSource1;
119 if (regSource2)
120 node->uses[useIdx++] = regSource2;
121}

163{
164 // Do not insert if the block is already inserted in the list of predecessors.

166 block->pred = listInsert(block->pred, pred, -1);

234void deleteCFG(t_cfg *graph)
235{
236 t_listNode *curNode;
237 if (graph == NULL)
238 return;
239
240 curNode = graph->blocks;
241 while (curNode) {
242 t_basicBlock *curBlock = (t_basicBlock *)curNode->data;
243 deleteBasicBlock(curBlock);
244 curNode = curNode->next;
245 }
246 deleteList(graph->blocks);
247 deleteBasicBlock(graph->endingBlock);
248
249 curNode = graph->registers;
250 while (curNode) {
251 t_cfgReg *curReg = (t_cfgReg *)curNode->data;
252 deleteList(curReg->mcRegWhitelist);
253 free(curReg);
254 curNode = curNode->next;
255 }
256 deleteList(graph->registers);
257
258 free(graph);

272static t_basicBlock *cfgSearchLabel(t_cfg *graph, t_label *label)
273{
274 if (label == NULL)
275 return NULL;
276
277 t_basicBlock *bblock = NULL;
278 t_listNode *curNode = graph->blocks;
279 while (curNode != NULL) {
280 bblock = (t_basicBlock *)curNode->data;
281
282 // Check the first node of the basic block. If its instruction has a label,
283 // check if it's the label we are searching for.
284 t_bbNode *curCFGNode = (t_bbNode *)bblock->nodes->data;
285 if ((curCFGNode->instr)->label != NULL) {
286 if ((curCFGNode->instr)->label->labelID == label->labelID)
287 // Found!
288 break;
289 }
290
291 curNode = curNode->next;
292 }
293
294 return bblock;
295}

308static void cfgComputeTransitions(t_cfg *graph)
309{
310 // This function is the continuation of programToCFG(), where after creating
311 // the blocks in the CFG we need to construct the transitions between them.
312 // After the basic block construction, all branch instructions are now
313 // found at the end of (some of the) basic blocks. The algorithm for adding
314 // the transitions simply consists of searching for every branch, and adding
315 // the correct outgoing edges to its basic block.
316 t_listNode *curNode = graph->blocks;
317 while (curNode != NULL) {
318 t_basicBlock *curBlock = (t_basicBlock *)curNode->data;
319
320 // Get the last instruction in the basic block.
321 t_listNode *lastNode = listGetLastNode(curBlock->nodes);
322 t_bbNode *lastCFGNode = (t_bbNode *)lastNode->data;
323 t_instruction *lastInstr = lastCFGNode->instr;
324
325 // If the instruction is return-like or exit-like, by definition the next
326 // block is the ending block because it stops the program/subroutine.
327 if (isExitInstruction(lastInstr)) {
328 bbAddSucc(curBlock, graph->endingBlock);
329 bbAddPred(graph->endingBlock, curBlock);
330 continue;
331 }
332
333 // All branch/jump instructions may transfer control to the code
334 // indicated by their label argument, so add edges appropriately.
335 if (isJumpInstruction(lastInstr)) {
336 if (lastInstr->addressParam == NULL)
337 fatalError("bug: malformed jump instruction with no label in CFG");
338 t_basicBlock *jumpBlock = cfgSearchLabel(graph, lastInstr->addressParam);
339 if (jumpBlock == NULL)
340 fatalError("bug: malformed jump instruction with invalid label in CFG");
341 bbAddPred(jumpBlock, curBlock);
342 bbAddSucc(curBlock, jumpBlock);
343 }
344
345 // Additionally, conditional jumps may also not be taken, and in that
346 // case the execution continues to the next instruction.
347 // As the order of the blocks in the block list reflects the order of
348 // the instructions in the program, we can rely on this property to fetch
349 // the correct block for this fallthrough case.
350 if (!isUnconditionalJump(lastInstr)) {
351 t_listNode *nextNode = curNode->next;
352 if (nextNode != NULL) {
353 // The current basic block has a successor in the list, all is fine
354 t_basicBlock *nextBlock = nextNode->data;
355 bbAddSucc(curBlock, nextBlock);
356 bbAddPred(nextBlock, curBlock);
357 } else {
358 // If this is the last basic block in the list, the next block is
359 // the ending block (which exists outside the list).
360 bbAddSucc(curBlock, graph->endingBlock);
361 bbAddPred(graph->endingBlock, curBlock);
362 }
363 }
364
365 curNode = curNode->next;
366 }
367}

369t_cfg *programToCFG(t_program *program)
370{
371 t_cfg *result = newCFG();
372
373 // Generate each basic block, by splitting the list of instruction at each
374 // terminator instruction or labeled instruction. Labeled instructions are
375 // instructions with a label assigned to them. Terminator instructions are
376 // branch-like instructions: branches themselves, but also any instruction
377 // used for subroutine return.
378 // The `bblock' variable contains a basic block we are adding instructions
379 // to, or NULL if we have just found the end of the last basic block and we
380 // are not sure whether to insert a new one. When `bblock' is NULL, a new
381 // block is created lazily at the next instruction found. This ensures no
382 // empty blocks are created.
383 t_basicBlock *bblock = NULL;
384 t_listNode *curNode = program->instructions;
385 while (curNode != NULL) {
386 t_instruction *curInstr = (t_instruction *)curNode->data;
387
388 // If the instruction node needs to be at the beginning of a basic block
389 // (= is labeled) or if `bblock' is NULL (because the last instruction was
390 // a terminator) then create a new basic block.
391 if (instrIsStartingNode(curInstr) || bblock == NULL)
392 bblock = cfgCreateBlock(result);
393
394 // Add the instruction to the end of the current basic block.
395 t_bbNode *curCFGNode = bbInsertInstruction(bblock, curInstr);
396
397 // If the instruction is a basic block terminator, set `bblock' to NULL
398 // to stop inserting nodes into it.
399 if (instrIsEndingNode(curInstr))
400 bblock = NULL;
401
402 curNode = curNode->next;
403 }
404
405 // Now all the blocks have been created, we need to add the edges between
406 // blocks, which is done in the cfgComputeTransitions() function.
407 cfgComputeTransitions(result);
408 return result;
409}

412void cfgToProgram(t_program *program, t_cfg *graph)
413{
414 // Erase the old code segment.
415 program->instructions = deleteList(program->instructions);
416
417 // Iterate through all the instructions in all the basic blocks (in order)
418 // and re-add them to the program.
419 t_listNode *curBlockNode = graph->blocks;
420 while (curBlockNode != NULL) {
421 t_basicBlock *bblock = (t_basicBlock *)curBlockNode->data;
422 t_listNode *curInnerNode = bblock->nodes;
423 while (curInnerNode != NULL) {
424 t_bbNode *node = (t_bbNode *)curInnerNode->data;
425
426 program->instructions =
427 listInsert(program->instructions, node->instr, -1);
428
429 curInnerNode = curInnerNode->next;
430 }
431 curBlockNode = curBlockNode->next;
432 }
433}

436int cfgIterateNodes(t_cfg *graph, void *context,
437 int (*callback)(t_bbNode *node, int nodeIndex, void *context))
438{
439 int counter = 0;
440 int exitcode = 0;
441
442 t_listNode *curBlockNode = graph->blocks;
443 while (curBlockNode != NULL) {
444 t_basicBlock *curBlock = (t_basicBlock *)curBlockNode->data;
445
446 t_listNode *curInnerNode = curBlock->nodes;
447 while (curInnerNode != NULL) {
448 t_bbNode *curCFGNode = (t_bbNode *)curInnerNode->data;
449
450 exitcode = callback(curCFGNode, counter, context);
451 if (exitcode != 0)
452 return exitcode;
453
454 counter++;
455 curInnerNode = curInnerNode->next;
456 }
457
458 curBlockNode = curBlockNode->next;
459 }
460 return exitcode;
461}

487static t_listNode *addElementToSet(t_listNode *set, void *element,
488 bool (*compareFunc)(void *a, void *b), bool *modified)
489{
490 if (element == NULL)
491 return set;
492
493 // Add the element if it's not already in the `set' list.
494 if (listFindWithCallback(set, element, compareFunc) == NULL) {
495 set = listInsert(set, element, -1);
496 if (modified != NULL)
497 (*modified) = true;
498 }
499
500 return set;
501}

503static t_listNode *addElementsToSet(t_listNode *set, t_listNode *elements,
504 bool (*compareFunc)(void *a, void *b), bool *modified)
505{
506 // Add all the elements to the set one by one.
507 t_listNode *curNode = elements;
508 while (curNode != NULL) {
509 void *curData = curNode->data;
510 set = addElementToSet(set, curData, compareFunc, modified);
511 curNode = curNode->next;
512 }
513
514 // Return the new list.
515 return set;
516}

518static t_listNode *computeLiveInSetEquation(t_cfgReg *defs[CFG_MAX_DEFS],
519 t_cfgReg *uses[CFG_MAX_USES], t_listNode *liveOut)
520{
521 // Initialize live in set as equal to live out set.
522 t_listNode *liveIn = listClone(liveOut);
523
524 // Add all items from set of uses.
525 for (int i = 0; i < CFG_MAX_USES; i++) {
526 if (uses[i] == NULL)
527 continue;
528 if (TARGET_REG_ZERO_IS_CONST && uses[i]->tempRegID == REG_0)
529 continue;
530 liveIn = addElementToSet(liveIn, uses[i], NULL, NULL);
531 }
532
533 // Remove items from set of definitions as long as they are not present in
534 // the set of uses.
535 for (int defIdx = 0; defIdx < CFG_MAX_DEFS; defIdx++) {
536 int found = 0;
537
538 if (defs[defIdx] == NULL)
539 continue;
540 if (TARGET_REG_ZERO_IS_CONST && defs[defIdx]->tempRegID == REG_0)
541 continue;
542
543 for (int useIdx = 0; useIdx < CFG_MAX_USES && !found; useIdx++) {
544 if (uses[useIdx] && uses[useIdx]->tempRegID == defs[defIdx]->tempRegID)
545 found = 1;
546 }
547
548 if (!found)
549 liveIn = listFindAndRemove(liveIn, defs[defIdx]);
550 }
551
552 return liveIn;
553}

555/* Re-computes the live registers out of a block by applying the standard
556 * flow equation:
557 * out(block) = union in(block') for all successor block' */
558static t_listNode *cfgComputeLiveOutOfBlock(t_cfg *graph, t_basicBlock *block)
559{
560 // Iterate through all the successor blocks
561 t_listNode *result = NULL;
562 t_listNode *curSuccNode = block->succ;
563 while (curSuccNode != NULL) {
564 t_basicBlock *curSuccessor = (t_basicBlock *)curSuccNode->data;
565
566 if (curSuccessor != graph->endingBlock) {
567 // Update our block's 'out' set by adding all registers 'in' to the
568 // current successor.
569 t_listNode *liveINRegs = bbGetLiveIn(curSuccessor);
570 result = addElementsToSet(result, liveINRegs, NULL, NULL);
571 deleteList(liveINRegs);
572 }
573
574 curSuccNode = curSuccNode->next;
575 }
576
577 return result;
578}

580static bool cfgUpdateLivenessOfNodesInBlock(t_cfg *graph, t_basicBlock *bblock)
581{
582 // Keep track of whether we modified something or not.
583 bool modified = false;
584
585 // Start with the last node in the basic block, we will proceed upwards
586 // from there.
587 t_listNode *curLI = listGetLastNode(bblock->nodes);
588 // The live in set of the successors of the last node in the block is the
589 // live out set of the block itself.
590 t_listNode *successorsLiveIn = cfgComputeLiveOutOfBlock(graph, bblock);
591
592 while (curLI != NULL) {
593 // Get the current CFG node.
594 t_bbNode *curNode = (t_bbNode *)curLI->data;
595
596 // Live out of a block is equal to the union of the live in sets of the
597 // successors.
598 curNode->out =
599 addElementsToSet(curNode->out, successorsLiveIn, NULL, &modified);
600 deleteList(successorsLiveIn);
601
602 // Compute the live in set of the block using the set of definition,
603 // uses and live out registers of the block.
604 t_listNode *liveIn =
605 computeLiveInSetEquation(curNode->defs, curNode->uses, curNode->out);
606 curNode->in = addElementsToSet(curNode->in, liveIn, NULL, &modified);
607
608 // Since there are no branches in a basic block, the successors of
609 // the predecessor is just the current block.
610 successorsLiveIn = liveIn;
611
612 // Continue backwards to the previous node.
613 curLI = curLI->prev;
614 }
615
616 deleteList(successorsLiveIn);
617
618 // Return a non-zero value if anything was modified.
619 return modified;
620}

622static bool cfgPerformLivenessIteration(t_cfg *graph)
623{
624 bool modified = false;
625 t_listNode *curNode = listGetLastNode(graph->blocks);
626 while (curNode != NULL) {
627 t_basicBlock *curBlock = (t_basicBlock *)curNode->data;
628
629 // Update the liveness informations for the current basic block.
630 if (cfgUpdateLivenessOfNodesInBlock(graph, curBlock))
631 modified = true;
632
633 curNode = curNode->prev;
634 }
635 return modified;
636}

643 } while (modified);
