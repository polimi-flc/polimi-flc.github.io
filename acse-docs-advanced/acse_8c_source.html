24char *getLogFileName(const char *logType, const char *fn)
25{
26 char *basename = strdup(fn);
27 if (!basename)
28 fatalError("out of memory");
29
30 int lastDot = -1;
31 for (int i = 0; basename[i] != '\0'; i++) {
32 if (basename[i] == '.')
33 lastDot = i;
34 }
35 if (lastDot >= 0)
36 basename[lastDot] = '\0';
37
38 size_t nameLen = strlen(basename) + strlen(logType) + (size_t)8;
39 char *outfn = calloc(nameLen, sizeof(char));
40 if (!outfn)
41 fatalError("out of memory");
42
43 snprintf(outfn, nameLen, "%s_%s.log", basename, logType);
44 free(basename);
45 return outfn;
46}
