183void setRawLabelName(t_program *program, t_label *label, const char *finalName)
184{
185 t_listNode *i;
186
187 // Check the entire list of labels because there might be two
188 // label objects with the same ID and they need to be kept in sync.
189 for (i = program->labels; i != NULL; i = i->next) {
190 t_label *thisLab = i->data;
191
192 if (thisLab->labelID == label->labelID) {
193 // Found! Remove old name.
194 free(thisLab->name);
195 // Change to new name.
196 if (finalName)
197 thisLab->name = strdup(finalName);
198 else
199 thisLab->name = NULL;
200 }
201 }
202}

204void setLabelName(t_program *program, t_label *label, const char *name)
205{
206 // Remove all non a-zA-Z0-9_ characters.
207 char *sanitizedName = calloc(strlen(name) + 1, sizeof(char));
208 if (!sanitizedName)
209 fatalError("out of memory");
210 const char *srcp = name;
211 for (char *dstp = sanitizedName; *srcp; srcp++) {
212 if (*srcp == '_' || isalnum(*srcp))
213 *dstp++ = *srcp;
214 }
215
216 // Append a sequential number to disambiguate labels with the same name.
217 size_t allocatedSpace = strlen(sanitizedName) + 24;
218 char *finalName = calloc(allocatedSpace, sizeof(char));
219 if (!finalName)
220 fatalError("out of memory");
221 snprintf(finalName, allocatedSpace, "%s", sanitizedName);
222 int serial = -1;
223 bool ok;
224 do {
225 t_listNode *i;
226 ok = true;
227 for (i = program->labels; i != NULL; i = i->next) {
228 t_label *thisLab = i->data;
229 char *thisLabName;
230 int difference;
231
232 if (thisLab->labelID == label->labelID)
233 continue;
234
235 thisLabName = getLabelName(thisLab);
236 difference = strcmp(finalName, thisLabName);
237 free(thisLabName);
238
239 if (difference == 0) {
240 ok = false;
241 snprintf(finalName, allocatedSpace, "%s_%d", sanitizedName, ++serial);
242 break;
243 }
244 }
245 } while (!ok);
246
247 free(sanitizedName);
248 setRawLabelName(program, label, finalName);
249 free(finalName);
250}

252void assignLabel(t_program *program, t_label *label)
253{
254 // Check if this label has already been assigned.
255 for (t_listNode *li = program->instructions; li != NULL; li = li->next) {
256 t_instruction *instr = li->data;
257 if (instr->label && instr->label->labelID == label->labelID)
258 fatalError("bug: label already assigned");
259 }
260
261 // Test if the next instruction already has a label.
262 if (program->pendingLabel != NULL) {
263 // It does: transform the label being assigned into an alias of the
264 // label of the next instruction's label.
265 // All label aliases have the same ID and name.
266
267 // Decide the name of the alias. If only one label has a name, that name
268 // wins. Otherwise the name of the label with the lowest ID wins.
269 char *name = program->pendingLabel->name;
270 if (!name ||
271 (label->labelID && label->labelID < program->pendingLabel->labelID))
272 name = label->name;
273 // Copy the name.
274 if (name)
275 name = strdup(name);
276
277 // Change ID and name.
278 label->labelID = (program->pendingLabel)->labelID;
279 setRawLabelName(program, label, name);
280
281 // Promote both labels to global if at least one is global.
282 if (label->global)
283 program->pendingLabel->global = 1;
284 else if (program->pendingLabel->global)
285 label->global = true;
286
287 // Mark the label as an alias.
288 label->isAlias = true;
289
290 free(name);
291 } else {
292 program->pendingLabel = label;
293 }
294}

311void addInstruction(t_program *program, t_instruction *instr)
312{
313 static t_fileLocation lastFileLoc = {NULL, -1};
314
315 // Assign the currently pending label if there is one.
316 instr->label = program->pendingLabel;
317 program->pendingLabel = NULL;
318
319 // Add a comment with the line number.
320 if (curFileLoc.row >= 0 &&
321 (curFileLoc.file != lastFileLoc.file ||
322 curFileLoc.row != lastFileLoc.row)) {
323 size_t fileNameLen = strlen(curFileLoc.file);
324 size_t strBufSz = fileNameLen + 10 + 1;
325 instr->comment = calloc(strBufSz, sizeof(char));
326 if (instr->comment) {
327 snprintf(instr->comment, strBufSz, "%s:%d", curFileLoc.file,
328 curFileLoc.row + 1);
329 }
330 }
331 lastFileLoc = curFileLoc;
332
333 // Update the list of instructions.
334 program->instructions = listInsert(program->instructions, instr, -1);
335}

337t_instruction *genInstruction(t_program *program, int opcode, t_regID rd,
338 t_regID rs1, t_regID rs2, t_label *label, int immediate)
339{
340 t_instruction *instr = newInstruction(opcode);
341 if (rd != REG_INVALID)
342 instr->rDest = newInstrArg(rd);
343 if (rs1 != REG_INVALID)
344 instr->rSrc1 = newInstrArg(rs1);
345 if (rs2 != REG_INVALID)
346 instr->rSrc2 = newInstrArg(rs2);
347 if (label)
348 instr->addressParam = label;
349 instr->immediate = immediate;
350
351 // Add the newly created instruction to the current program. This function
352 // may be called with program set to NULL to just create the instruction
353 // without inserting it, so we need to handle this case as well.
354 if (program != NULL)
355 addInstruction(program, instr);
356
357 return instr;
358}

360void removeInstructionAt(t_program *program, t_listNode *instrLi)
361{
362 t_instruction *instrToRemove = (t_instruction *)instrLi->data;
363
364 // Move the label and/or the comment to the next instruction.
365 if (instrToRemove->label || instrToRemove->comment) {
366 // Find the next instruction, if it exists.
367 t_listNode *nextPos = instrLi->next;
368 t_instruction *nextInst = NULL;
369 if (nextPos)
370 nextInst = nextPos->data;
371
372 // Move the label.
373 if (instrToRemove->label) {
374 // Generate a nop if there was no next instruction or if the next
375 // instruction is already labeled.
376 if (!nextInst || (nextInst->label)) {
377 nextInst = genNOP(NULL);
378 program->instructions =
379 listInsertAfter(program->instructions, instrLi, nextInst);
380 }
381 nextInst->label = instrToRemove->label;
382 instrToRemove->label = NULL;
383 }
384
385 // Move the comment, if possible; otherwise it will be discarded.
386 if (nextInst && instrToRemove->comment && !nextInst->comment) {
387 nextInst->comment = instrToRemove->comment;
388 instrToRemove->comment = NULL;
389 }
390 }
391
392 // Remove the instruction.
393 program->instructions = listRemoveNode(program->instructions, instrLi);
394 deleteInstruction(instrToRemove);
395}

405t_symbol *createSymbol(
406 t_program *program, char *ID, t_symbolType type, int arraySize)
407{
408 // Check validity of type.
409 if (type != TYPE_INT && type != TYPE_INT_ARRAY)
410 fatalError("bug: invalid type");
411 // Check array size validity.
412 if (type == TYPE_INT_ARRAY && arraySize <= 0) {
413 emitError(curFileLoc, "invalid size %d for array %s", arraySize, ID);
414 return NULL;
415 }
416
417 // Check if another symbol already exists with the same ID.
418 t_symbol *existingSym = getSymbol(program, ID);
419 if (existingSym != NULL) {
420 emitError(curFileLoc, "variable '%s' already declared", ID);
421 return NULL;
422 }
423
424 // Allocate and initialize a new symbol object.
425 t_symbol *res = newSymbol(ID, type, arraySize);
426
427 // Reserve a new label for the variable.
428 res->label = createLabel(program);
429
430 // Set the name of the label.
431 char *lblName = calloc(strlen(ID) + 8, sizeof(char));
432 if (!lblName)
433 fatalError("out of memory");
434 sprintf(lblName, "l_%s", ID);
435 setLabelName(program, res->label, lblName);
436 free(lblName);
437
438 // Now we can add the new variable to the program.
439 program->symbols = listInsert(program->symbols, res, -1);
440 return res;
441}

492void dumpProgram(t_program *program, FILE *fout)
493{
494 fprintf(fout, "# Program dump\n\n");
495
496 fprintf(fout, "## Variables\n\n");
497 t_listNode *curVarNode = program->symbols;
498 while (curVarNode) {
499 t_symbol *var = curVarNode->data;
500 fprintf(fout, "\"%s\":\n", var->ID);
501
502 if (var->type == TYPE_INT) {
503 fprintf(fout, " type = int\n");
504 } else if (var->type == TYPE_INT_ARRAY) {
505 fprintf(fout, " type = int[%d]\n", var->arraySize);
506 } else {
507 fprintf(fout, " type = invalid\n");
508 }
509 char *labelName = getLabelName(var->label);
510 fprintf(fout, " label = %s (ID=%d)\n", labelName, var->label->labelID);
511 free(labelName);
512
513 curVarNode = curVarNode->next;
514 }
515
516 fprintf(fout, "\n## Instructions\n\n");
517 t_listNode *curInstNode = program->instructions;
518 while (curInstNode) {
519 t_instruction *instr = curInstNode->data;
520 if (instr == NULL)
521 fprintf(fout, "(null)");
522 else
523 printInstruction(instr, fout, false);
524 fprintf(fout, "\n");
525 curInstNode = curInstNode->next;
526 }
527
528 fflush(fout);
529}
