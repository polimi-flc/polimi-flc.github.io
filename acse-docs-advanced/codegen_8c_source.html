10void validateRegisterId(t_program *program, t_regID r)
11{
12 if (!program)
13 return;
14 if (r >= 0 && r < program->firstUnusedReg)
15 return;
16 fatalError("bug: invalid register identifier %d", r);
17}

441t_regID genLoadVariable(t_program *program, t_symbol *var)
442{
443 // Check if the symbol is an array; in that case do not generate any more
444 // code. Calling emitError will eventually stop compilation anyway.
445 if (isArray(var)) {
446 emitError(curFileLoc, "'%s' is an array", var->ID);
447 return REG_0;
448 }
449
450 // Generate an LA instruction to load the label address into a register.
451 t_regID rAddr = getNewRegister(program);
452 genLA(program, rAddr, var->label);
453 // Generate a LW from the address.
454 t_regID rRes = getNewRegister(program);
455 genLW(program, rRes, 0, rAddr);
456 return rRes;
457}

460void genStoreRegisterToVariable(t_program *program, t_symbol *var, t_regID reg)
461{
462 // Check if the symbol is an array; in that case bail out without generating
463 // any code (but emitting an error that will eventually stop further
464 // compilation).
465 if (isArray(var)) {
466 emitError(curFileLoc, "'%s' is an array", var->ID);
467 return;
468 }
469
470 // Generate an LA instruction to load the label address into a register.
471 t_regID rAddr = getNewRegister(program);
472 genLA(program, rAddr, var->label);
473 // Generate a SW to the address specified by the label.
474 genSW(program, reg, 0, rAddr);
475}

495t_regID genLoadArrayAddress(t_program *program, t_symbol *array, t_regID rIdx)
496{
497 if (!isArray(array)) {
498 // If the symbol is not an array, bail out returning a dummy register ID.
499 emitError(curFileLoc, "'%s' is a scalar", array->ID);
500 return REG_0;
501 }
502 t_label *label = array->label;
503
504 // Generate a load of the base address using LA
505 t_regID rAddr = getNewRegister(program);
506 genLA(program, rAddr, label);
507
508 // Generate the code to compute the offset of the element in the array in
509 // bytes. Assume the type is an integer (no other scalar types are supported).
510 t_regID rOffset;
511 int sizeofElem = 4 / TARGET_PTR_GRANULARITY;
512 if (sizeofElem != 1) {
513 rOffset = getNewRegister(program);
514 genMULI(program, rOffset, rIdx, sizeofElem);
515 } else {
516 rOffset = rIdx;
517 }
518
519 // Generate the code which computes the final address by summing the base
520 // address to the offset of the element.
521 genADD(program, rAddr, rAddr, rOffset);
522 return rAddr;
523}

526t_regID genLoadArrayElement(t_program *program, t_symbol *array, t_regID rIdx)
527{
528 // Generate code that loads the address of the array element in a register
529 // and then loads the element itself from that memory address.
530 t_regID rAddr = genLoadArrayAddress(program, array, rIdx);
531 t_regID rVal = getNewRegister(program);
532 genLW(program, rVal, 0, rAddr);
533 return rVal;
