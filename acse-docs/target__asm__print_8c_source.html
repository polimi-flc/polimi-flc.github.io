312int instructionToString(
313 char *buf, int bufsz, t_instruction *instr, bool machineRegIDs)
314{
315 int res;
316 char *buf0 = buf;
317 char *address = NULL;
318
319 const char *opc = opcodeToString(instr->opcode);
320 char *rd = registerToString(instr->rDest, machineRegIDs);
321 char *rs1 = registerToString(instr->rSrc1, machineRegIDs);
322 char *rs2 = registerToString(instr->rSrc2, machineRegIDs);
323 if (instr->addressParam) {
324 int n = labelToString(NULL, 0, instr->addressParam, 0);
325 address = calloc((size_t)n + 1, sizeof(char));
326 if (!address)
327 fatalError("out of memory");
328 labelToString(address, n + 1, instr->addressParam, 0);
329 }
330 int32_t imm = instr->immediate;
331
332 int format = opcodeToFormat(instr->opcode);
333 switch (format) {
334 case FORMAT_OP:
335 if (!instr->rDest || !instr->rSrc1 || !instr->rSrc2)
336 fatalError("bug: invalid instruction found in the program");
337 res = snprintf(buf, bufsz, "%-6s %s, %s, %s", opc, rd, rs1, rs2);
338 break;
339 case FORMAT_OPIMM:
340 if (!instr->rDest || !instr->rSrc1)
341 fatalError("bug: invalid instruction found in the program");
342 res = snprintf(buf, bufsz, "%-6s %s, %s, %d", opc, rd, rs1, imm);
343 break;
344 case FORMAT_LOAD:
345 if (!instr->rDest || !instr->rSrc1)
346 fatalError("bug: invalid instruction found in the program");
347 res = snprintf(buf, bufsz, "%-6s %s, %d(%s)", opc, rd, imm, rs1);
348 break;
349 case FORMAT_LOAD_GL:
350 if (!instr->rDest || !instr->addressParam)
351 fatalError("bug: invalid instruction found in the program");
352 res = snprintf(buf, bufsz, "%-6s %s, %s", opc, rd, address);
353 break;
354 case FORMAT_STORE:
355 if (!instr->rSrc2 || !instr->rSrc1)
356 fatalError("bug: invalid instruction found in the program");
357 res = snprintf(buf, bufsz, "%-6s %s, %d(%s)", opc, rs2, imm, rs1);
358 break;
359 case FORMAT_STORE_GL:
360 if (!instr->rDest || !instr->rSrc1 || !instr->addressParam)
361 fatalError("bug: invalid instruction found in the program");
362 res = snprintf(buf, bufsz, "%-6s %s, %s, %s", opc, rs1, address, rd);
363 break;
364 case FORMAT_BRANCH:
365 if (!instr->rSrc1 || !instr->rSrc2 || !instr->addressParam)
366 fatalError("bug: invalid instruction found in the program");
367 res = snprintf(buf, bufsz, "%-6s %s, %s, %s", opc, rs1, rs2, address);
368 break;
369 case FORMAT_JUMP:
370 if (!instr->addressParam)
371 fatalError("bug: invalid instruction found in the program");
372 res = snprintf(buf, bufsz, "%-6s %s", opc, address);
373 break;
374 case FORMAT_LI:
375 if (!instr->rDest)
376 fatalError("bug: invalid instruction found in the program");
377 res = snprintf(buf, bufsz, "%-6s %s, %d", opc, rd, imm);
378 break;
379 case FORMAT_LA:
380 if (!instr->rDest || !instr->addressParam)
381 fatalError("bug: invalid instruction found in the program");
382 res = snprintf(buf, bufsz, "%-6s %s, %s", opc, rd, address);
383 break;
384 case FORMAT_SYSTEM:
385 res = snprintf(buf, bufsz, "%s", opc);
386 break;
387 case FORMAT_FUNC:
388 default:
389 if (instr->rDest)
390 buf += sprintf(buf, "%s = ", rd);
391 buf += sprintf(buf, "%s(", opc);
392 if (instr->rSrc1)
393 buf += sprintf(buf, "%s", rs1);
394 if (instr->rSrc1 && instr->rSrc2)
395 buf += sprintf(buf, ", ");
396 if (instr->rSrc2)
397 buf += sprintf(buf, "%s", rs2);
398 buf += sprintf(buf, ")");
399 res = (int)(buf - buf0);
400 break;
401 }
402
403 free(address);
404 free(rd);
405 free(rs1);
406 free(rs2);
407 return res;
408}

411bool translateForwardDeclarations(t_program *program, FILE *fp)
412{
413 for (t_listNode *li = program->labels; li != NULL; li = li->next) {
414 t_label *nextLabel = li->data;
415
416 if (nextLabel->isAlias)
417 continue;
418
419 if (nextLabel->global) {
420 char *labelName;
421 int res;
422
423 labelName = getLabelName(nextLabel);
424 res = fprintf(fp, "%-8s.global %s\n", "", labelName);
425 free(labelName);
426
427 if (res < 0)
428 return false;
429 }
430 }
431
432 return true;
433}

436bool printInstruction(t_instruction *instr, FILE *fp, bool machineRegIDs)
437{
438 char buf[BUF_LENGTH];
439
440 if (instr->label != NULL) {
441 labelToString(buf, BUF_LENGTH, instr->label, 1);
442 } else {
443 buf[0] = '\0';
444 }
445 if (fprintf(fp, "%-8s", buf) < 0)
446 return false;
447
448 instructionToString(buf, BUF_LENGTH, instr, machineRegIDs);
449 int res;
450 if (instr->comment) {
451 res = fprintf(fp, "%-48s# %s", buf, instr->comment);
452 } else {
453 res = fprintf(fp, "%s", buf);
454 }
455 if (res < 0)
456 return false;
457
458 return true;
459}

461bool translateCodeSegment(t_program *program, FILE *fp)
462{
463 if (!program->instructions)
464 return true;
465
466 // Write the .text directive to switch to the text segment.
467 if (fprintf(fp, "%-8s.text\n", "") < 0)
468 return false;
469
470 t_listNode *curNode = program->instructions;
471 while (curNode != NULL) {
472 t_instruction *curInstr = (t_instruction *)curNode->data;
473 if (curInstr == NULL)
474 fatalError("bug: NULL instruction found in the program");
475
476 if (!printInstruction(curInstr, fp, true))
477 return false;
478 if (fprintf(fp, "\n") < 0)
479 return false;
480
481 curNode = curNode->next;
482 }
483 return true;
484}

487int printGlobalDeclaration(t_symbol *data, FILE *fp)
488{
489 char buf[BUF_LENGTH];
490
491 // Print the label.
492 if (data->label != NULL) {
493 labelToString(buf, BUF_LENGTH, data->label, 1);
494 } else {
495 buf[0] = '\0';
496 }
497 if (fprintf(fp, "%-8s", buf) < 0)
498 return -1;
499
500 // Print the directive.
501 int size;
502 switch (data->type) {
503 case TYPE_INT:
504 size = 4 / TARGET_PTR_GRANULARITY;
505 break;
506 case TYPE_INT_ARRAY:
507 size = (4 / TARGET_PTR_GRANULARITY) * data->arraySize;
508 break;
509 default:
510 fatalError("bug: invalid data type found in the program");
511 }
512 if (fprintf(fp, ".space %d", size) < 0)
513 return -1;
514
515 return 0;
516}

518bool translateDataSegment(t_program *program, FILE *fp)
519{
520 // If the symbol table is empty, nothing to do.
521 if (program->symbols == NULL)
522 return true;
523
524 // Write the .data directive to switch to the data segment.
525 if (fprintf(fp, "%-8s.data\n", "") < 0)
526 return false;
527
528 // Print a static declaration for each symbol.
529 t_listNode *li = program->symbols;
530 while (li != NULL) {
531 t_symbol *symbol = (t_symbol *)li->data;
532
533 if (printGlobalDeclaration(symbol, fp) < 0)
534 return false;
535 if (fprintf(fp, "\n") < 0)
536 return false;
537
538 li = li->next;
539 }
540
541 return true;
542}

545bool writeAssembly(t_program *program, const char *fn)
546{
547 bool res = false;
548 FILE *fp = fopen(fn, "w");
549 if (fp == NULL)
550 return res;
551
552 if (!translateForwardDeclarations(program, fp))
553 goto fail;
554 if (!translateDataSegment(program, fp))
555 goto fail;
556 if (!translateCodeSegment(program, fp))
557 goto fail;
558
559 res = true;
560fail:
561 if (fclose(fp) == EOF)
562 res = false;
563 return res;
564}
