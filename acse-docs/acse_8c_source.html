85char *getLogFileName(const char *logType, const char *fn)
86{
87 char *basename = strdup(fn);
88 if (!basename)
89 fatalError("out of memory");
90
91 int lastDot = -1;
92 for (int i = 0; basename[i] != '\0'; i++) {
93 if (basename[i] == '.')
94 lastDot = i;
95 }
96 if (lastDot >= 0)
97 basename[lastDot] = '\0';
98
99 size_t nameLen = strlen(basename) + strlen(logType) + (size_t)8;
100 char *outfn = calloc(nameLen, sizeof(char));
101 if (!outfn)
102 fatalError("out of memory");
103
104 snprintf(outfn, nameLen, "%s_%s.log", basename, logType);
105 free(basename);
106 return outfn;
107}
