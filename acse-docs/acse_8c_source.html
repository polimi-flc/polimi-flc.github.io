56char *getLogFileName(const char *logType, const char *fn)
57{
58 char *basename = strdup(fn);
59 if (!basename)
60 fatalError("out of memory");
61
62 int lastDot = -1;
63 for (int i = 0; basename[i] != '\0'; i++) {
64 if (basename[i] == '.')
65 lastDot = i;
66 }
67 if (lastDot >= 0)
68 basename[lastDot] = '\0';
69
70 size_t nameLen = strlen(basename) + strlen(logType) + (size_t)8;
71 char *outfn = calloc(nameLen, sizeof(char));
72 if (!outfn)
73 fatalError("out of memory");
74
75 snprintf(outfn, nameLen, "%s_%s.log", basename, logType);
76 free(basename);
77 return outfn;
78}
